272 static std::size_t saInputDeviceRead(snd_pcm_t* handle,
273 saInputSource::saSourceType* buffer,
274 const std::size_t framesToRead,
275 bool& isOverRun)
276 {
277 long numRead = snd_pcm_readi(handle, buffer, framesToRead);
278
279 int err = (int)numRead;
280 if (err == -EPIPE)
281 {
282 numRead = 0;
283 isOverRun = true;
284 }
285 else if (err == -EBADFD)
286 {
287 numRead = 0;
288 throw std::runtime_error
289 ( std::string("audio input device read failed: ") +
290 std::string("PCM is not in the right state") );
291 }
292 else if (err == -ESTRPIPE)
293 {
294 numRead = 0;
295 throw std::runtime_error
296 ( std::string("audio input device read failed: ") +
297 std::string("a suspend event occurred") );
298 }
299 else if (err < 0)
300 {
301 numRead = 0;
302 throw std::runtime_error
303 ( std::string("audio input device read failed: ") +
304 std::string(snd_strerror(err)));
305 }
306
307 return static_cast<std::size_t>(numRead);
308 }
