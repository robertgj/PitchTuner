173 std::size_t saInputStreamImpl::Read( std::vector< saSample >& buffer,
174 const std::size_t framesToRead )
175 {
176 // Data available? Try to fulfill a request
177 std::size_t samplesRemainingInFragment = 0;
178 std::size_t samplesReadFromFragment = 0;
179 saInputSource::saSourceType* nextSourceInput = 0;
180 if (currentFragment != 0)
181 {
182 samplesRemainingInFragment = samplesPerFrame*framesRemainingInFragment;
183 samplesReadFromFragment = samplesPerFrame*framesReadFromFragment;
184 nextSourceInput = &(currentFragment[samplesReadFromFragment]);
185 }
186
187 // Read. Collect a fragment from the device. Copy samples from the fragment
188 // into the buffer. As the samples are copied they are are transformed
189 // from saSourceType to saSample. Release the fragment and Collect another.
190 std::size_t samplesRead = 0;
191 std::size_t samplesToRead = samplesPerFrame*framesToRead;
192 while (samplesToRead > 0)
193 {
194 if (samplesRemainingInFragment >= samplesToRead)
195 {
196 std::transform(nextSourceInput,
197 nextSourceInput+samplesToRead,
198 std::back_inserter(buffer),
199 saSampleFromSource<saInputSource::saSourceType>());
200 nextSourceInput += samplesToRead;
201 samplesReadFromFragment += samplesToRead;
202 samplesRead += samplesToRead;
203 samplesRemainingInFragment -= samplesToRead;
204 samplesToRead = 0;
205 }
206 else if (samplesRemainingInFragment > 0)
207 {
208 std::transform(nextSourceInput,
209 nextSourceInput+samplesRemainingInFragment,
210 std::back_inserter(buffer),
211 saSampleFromSource<saInputSource::saSourceType>());
212 nextSourceInput += samplesRemainingInFragment;
213 samplesReadFromFragment += samplesRemainingInFragment;
214 samplesRead += samplesRemainingInFragment;
215 samplesToRead -= samplesRemainingInFragment;
216 samplesRemainingInFragment = 0;
217 }
218 else if ( EndOfSource() )
219 {
220 break;
221 }
222 else
223 {
224 if (currentFragment != 0)
225 {
226 inputSource->Release(currentFragment);
227 }
228 currentFragment = inputSource->Collect(framesThisFragment);
229 nextSourceInput = currentFragment;
230 samplesReadFromFragment = 0;
231 framesRemainingInFragment = framesThisFragment;
232 samplesRemainingInFragment =
233 samplesPerFrame*framesRemainingInFragment;
234 }
235
236 // Sanity checks
237 assert( (samplesRemainingInFragment+samplesReadFromFragment)
238 == samplesPerFrame*framesThisFragment );
239 assert( static_cast<std::size_t>(nextSourceInput-currentFragment)
240 == samplesReadFromFragment );
241 }
242
243 // Done
244 framesRemainingInFragment = samplesRemainingInFragment/samplesPerFrame;
245 framesReadFromFragment = samplesReadFromFragment/samplesPerFrame;
246 std::size_t framesRead = samplesRead/samplesPerFrame;
247 framesReadFromStream += framesRead;
248 return framesRead;
249 }

251 std::size_t saInputStreamImpl::Read( std::vector< saSample >& buffer,
252 std::size_t framesToRead,
253 const std::size_t sampleInFrame )
254 {
255 // Sanity check (sampleInFrame is zero-based)
256 if (sampleInFrame >= samplesPerFrame)
257 {
258 throw std::runtime_error("Illegal sampleInFrame");
259 }
260
261 // Ignore sampleInFrame argument for efficiency
262 if (samplesPerFrame == 1)
263 {
264 return Read( buffer, framesToRead );
265 }
266
267 // Data available? Try to fulfill a request
268 saInputSource::saSourceType* nextSourceInput = 0;
269 if (currentFragment != 0)
270 {
271 nextSourceInput =
272 &(currentFragment[samplesPerFrame*framesReadFromFragment]);
273 }
274
275 std::size_t framesRead = 0;
276 std::size_t framesLeftToRead = framesToRead;
277 while (framesLeftToRead>0)
278 {
279 if (framesRemainingInFragment > framesLeftToRead)
280 {
281 std::generate_n(std::back_inserter(buffer),
282 framesLeftToRead,
283 DecimateSource<saInputSource::saSourceType>
284 (nextSourceInput+sampleInFrame, samplesPerFrame));
285
286 nextSourceInput += samplesPerFrame*framesLeftToRead;
287 framesReadFromFragment += framesLeftToRead;
288 framesRead += framesLeftToRead;
289 framesRemainingInFragment -= framesLeftToRead;
290 framesLeftToRead = 0;
291 }
292 else if (framesRemainingInFragment > 0)
293 {
294 std::generate_n(std::back_inserter(buffer),
295 framesRemainingInFragment,
296 DecimateSource<saInputSource::saSourceType>
297 (nextSourceInput+sampleInFrame, samplesPerFrame));
298
299 nextSourceInput += samplesPerFrame*framesRemainingInFragment;
300 framesReadFromFragment += framesRemainingInFragment;
301 framesRead += framesRemainingInFragment;
302 framesLeftToRead -= framesRemainingInFragment;
303 framesRemainingInFragment = 0;
304 }
305 else if ( EndOfSource() )
306 {
307 break;
308 }
309 else
310 {
311 framesReadFromFragment = 0;
312 if ( currentFragment != 0 )
313 {
314 inputSource->Release(currentFragment);
315 }
316 currentFragment = inputSource->Collect(framesThisFragment);
317 framesRemainingInFragment = framesThisFragment;
318 nextSourceInput = currentFragment;
319 }
320
321 // Sanity checks
322 assert( (framesRemainingInFragment+framesReadFromFragment)
323 == framesThisFragment );
324 assert( static_cast<std::size_t>(nextSourceInput-currentFragment)
325 == samplesPerFrame*framesReadFromFragment );
326 }
327
328 // Done
329 framesReadFromStream += framesRead;
330 return framesRead;
331 }
