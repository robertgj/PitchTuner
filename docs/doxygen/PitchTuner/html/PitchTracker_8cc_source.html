336float PitchTrackerImpl::EstimatePitch(const std::deque<saSample>& y)
337{
338 // Sanity check
339 float pitch = 0;
340 voiced = false;
341 if (y.size() < WindowLength+MaxLags+1)
342 {
343 throw std::runtime_error("Not enough samples");
344 }
345
346 // Sample autocorrelation
347 std::vector<saSample> r(MaxLags+1,0);
348 r[0] = std::inner_product(y.rbegin(),
349 y.rbegin()+static_cast<long>(WindowLength),
350 y.rbegin(),
351 saSample(0));
352 if ( r[0] < threshold )
353 {
354 return pitch;
355 }
356
357 // Loop calculating autocorrelations
358 saSample dDC0 =
359 std::accumulate( y.rbegin(),
360 y.rbegin()+static_cast<long>(WindowLength),
361 saSample(0) );
362 saSample dDC = 0;
363 saSample dT = 0;
364 saSample dTsum = 0;
365 saSample cdT = 0;
366 float minT = 0;
367 float minCDT = 1;
368 bool minFound = false;
369 std::vector<saSample> vdT(MaxLags+1,0);
370 std::vector<saSample> vdDC(MaxLags+1,0);
371 std::vector<saSample> vcdT(MaxLags+1,0);
372 vcdT[0] = 1;
373 for (unsigned int T=1; T<=MaxLags; T++)
374 {
375 saSample t = static_cast<saSample>(T);
376
377 // Calculate this lag. Recall that the newest sample is at y.end()
378 r[T] = std::inner_product(y.rbegin(),
379 y.rbegin()+static_cast<long>(WindowLength),
380 y.rbegin()+static_cast<long>(T),
381 saSample(0));
382
383 // Calculate autocorrelation at this lag
384 saSample r0T = std::inner_product(y.rbegin()+T,
385 y.rbegin()+T+
386 static_cast<long>(WindowLength),
387 y.rbegin()+T,
388 saSample(0));
389
390 // Difference function
391 dT = r[0] + r0T - (2*r[T]);
392
393 // Remove DC component
394 // Test with:
395 // fs=48000; f0=480;
396 // t=0.25*sin(2*pi*(0:((fs/10)-1))*f0/fs)+(0.5*((0:((fs/10)-1))/(fs/10)));
397 // audiowrite("sin480HzDC.wav",t,fs);
398 if (removeDC)
399 {
400 dDC = dDC0 - std::accumulate(y.rbegin()+T,
401 y.rbegin()+T+
402 static_cast<long>(WindowLength),
403 saSample(0));
404 dDC *= dDC;
405 dDC /= static_cast<saSample>(WindowLength);
406 dT -= dDC;
407 }
408
409 // Cumulative difference
410 dTsum += dT;
411 cdT = dTsum == 0 ? 0 : (dT*t)/dTsum;
412 vcdT[T] = cdT;
413
414 // Check for a minimum
415 if ( !minFound &&
416 (T>=2) &&
417 (vcdT[T-2] != 0) &&
418 (vcdT[T] != 0) &&
419 (vcdT[T-2] >= vcdT[T-1]) &&
420 (vcdT[T] >= vcdT[T-1]) )
421 {
422 // Parabolic interpolation
423 saSample c0 = vcdT[T-2];
424 saSample c1 = vcdT[T-1]-c0;
425 saSample c2 = ((vcdT[T]-c0)/2)-c1;
426 if (c2 == 0)
427 {
428 continue;
429 }
430
431 // Coefficients of p(x) = a*x^2 + b*x + c
432 saSample a = c2;
433 saSample b = c1 - (c2*((2*t)-3));
434 saSample c = c0 - ((t-2)*(c1-(c2*(t-1))));
435
436 // Find minimum value and period
437 saSampleTo<float> conv;
438 minT = -conv(b)/conv(2*a);
439 float minY = conv(c) - (conv(b*b)/conv(4*a));
440
441 // Store the minimum
442 if (minY < minCDT)
443 {
444 minCDT = minY;
445 }
446
447 // Pitch found?
448 if ( minY < threshold )
449 {
450 minFound = true;
451 voiced = true;
452 pitch = sampleRate/minT;
453 }
454 }
455
456 if (debug)
457 {
458 vdT[T] = dT;
459 if (removeDC)
460 {
461 vdDC[T] = dDC;
462 }
463 continue;
464 }
465 else if(minFound)
466 {
467 break;
468 }
469 }
470
471 // Done
472 if (debug)
473 {
474 dDCList.push_back(vdDC);
475 dtList.push_back(vdT);
476 cdtList.push_back(vcdT);
477 pitchList.push_back(pitch);
478 minCDTList.push_back(minCDT);
479 }
480
481 return pitch;
482}
