19void guiMeter::OnPaint( wxPaintEvent &WXUNUSED(event) )

67void guiMeter::DrawDial(wxMemoryDC &guiDialDC)
68{
69 // Clear bitmap
70 wxPen pen (*wxWHITE, 1);
71 guiDialDC.SetBrush(*wxWHITE_BRUSH);
72 guiDialDC.DrawRectangle(0, 0, m_w, m_h);
73
74 // Set the pen
75 pen.SetColour(*wxBLACK);
76 guiDialDC.SetPen(pen);
77
78 // Draw Units
79 guiDialDC.SetFont(fontUnits);
80 wxCoord unitsW, unitsH;
81 guiDialDC.GetTextExtent(m_units, &unitsW, &unitsH);
82 int unitsStartX = m_ptrOriginX - (0.5*unitsW);
83 int unitsStartY = m_ptrOriginY - (0.9*m_dialRadius) + (0.5*unitsH);
84 guiDialDC.DrawText(m_units, unitsStartX, unitsStartY);
85
86 // Set the ticks font
87 guiDialDC.SetFont(fontTicks);
88 guiDialDC.SetTextForeground(*wxBLACK);
89 guiDialDC.SetBackgroundMode(wxTRANSPARENT);
90
91 // Draw tick marks
92 int tickMajorAngle = 20;
93 int tickMediumAngle = 10;
94 int tickMinorAngle = 2;
95 int posText = 1.075*m_dialRadius;
96 int lenMajorTick = 1.05*m_dialRadius;
97 int lenMediumTick = 1.035*m_dialRadius;
98 int lenMinorTick = 1.02*m_dialRadius;
99 // Pointer angle from vertical
100 int angle;
101
102 // Draw tick marks
103 for(angle=-m_ptrMaxAngle; // Left hand side
104 angle<=m_ptrMaxAngle; // Right hand side
105 angle+=tickMinorAngle)
106 {
107 float sinAngle = sin(angle*pi/180);
108 float cosAngle = cos(angle*pi/180);
109
110 int startX = m_ptrOriginX+(sinAngle*m_dialRadius);
111 int startY = m_ptrOriginY-(cosAngle*m_dialRadius);
112
113 float tickRadius =
114 (angle%tickMajorAngle == 0 ? lenMajorTick :
115 (angle%tickMediumAngle == 0 ? lenMediumTick : lenMinorTick));
116
117 // DC origin is at top left corner
118 int endX = m_ptrOriginX+(sinAngle*tickRadius);
119 int endY = m_ptrOriginY-(cosAngle*tickRadius);
120
121 guiDialDC.DrawLine(startX, startY, endX, endY);
122 }
123
124 // Draw values
125 for(angle=-m_ptrMaxAngle; angle<=m_ptrMaxAngle; angle+=tickMajorAngle)
126 {
127 float sinAngle = sin(angle*pi/180);
128 float cosAngle = cos(angle*pi/180);
129
130 wxString text;
131 text.Printf(wxT("%d"), angle*m_ptrMaxError/m_ptrMaxAngle);
132
133 int textX = m_ptrOriginX+(sinAngle*posText);
134 int textY = m_ptrOriginY-(cosAngle*posText);
135
136 wxCoord textW, textH;
137 guiDialDC.GetTextExtent(text, &textW, &textH);
138
139 // Text and DC origins are at top left corner
140 int textStartX = textX - (0.5*textW*cosAngle) + (textH*sinAngle);
141 int textStartY = textY - (0.5*textW*sinAngle) - (textH*cosAngle);
142
143 // WindowsCE seems to have a problem drawing rotated text
144 // with a negative angle. To avoid this keep the angle in
145 // the range 0 to 360
146 int drawnAngle = (angle<0 ? -angle : 360-angle);
147 guiDialDC.DrawRotatedText(text, textStartX, textStartY, drawnAngle);
148 }
149}

151void guiMeter::DrawUpdate(wxAutoBufferedPaintDC &guiMeterDC)
152{
153 // Set the pen
154 wxPen pen (*wxBLACK, 1);
155 guiMeterDC.SetPen(pen);
156
157 // Set colour
158 guiMeterDC.SetTextForeground(*wxBLACK);
159 guiMeterDC.SetBackgroundMode(wxTRANSPARENT);
160
161 // Name
162 wxString blank = wxString(" ");
163 wxString textName(blank);
164 if ( m_voiced )
165 {
166 textName = m_name;
167 }
168 guiMeterDC.SetFont(fontName);
169 wxCoord nameW, nameH;
170 guiMeterDC.GetTextExtent(textName, &nameW, &nameH);
171 int nameStartX = (0.125*m_w) - (0.5*nameW) ;
172 int nameStartY = (0.8*m_h) - (0.5*nameH);
173 guiMeterDC.DrawText(textName, nameStartX, nameStartY);
174
175 // Frequency
176 wxString textFrequency(blank);
177 if ( m_voiced )
178 {
179 textFrequency.Printf(wxT("%3.1f"), m_frequency);
180 }
181 guiMeterDC.SetFont(fontFrequency);
182 wxCoord frequencyW, frequencyH;
183 guiMeterDC.GetTextExtent(textFrequency, &frequencyW, &frequencyH);
184 int frequencyStartX = (0.8*m_w) - (0.5*frequencyW) ;
185 int frequencyStartY = (0.8*m_h) - (0.5*frequencyH);
186 guiMeterDC.DrawText(textFrequency, frequencyStartX, frequencyStartY);
187
188 // Pitch
189 wxString textPitch(wxT("Not Voiced"));
190 if ( m_voiced )
191 {
192 textPitch.Printf(wxT("%4.1f"), m_pitch);
193 }
194 guiMeterDC.SetFont(fontPitch);
195 wxCoord pitchW, pitchH;
196 guiMeterDC.GetTextExtent(textPitch, &pitchW, &pitchH);
197 int pitchStartX = (0.5*m_w) - (0.5*pitchW) ;
198 int pitchStartY = (0.9*m_h) - (0.5*pitchH);
199 guiMeterDC.DrawText(textPitch, pitchStartX, pitchStartY);
200
201 // Pointer
202 if (m_voiced)
203 {
204 int angle = (m_pitchError*m_ptrMaxAngle)/m_ptrMaxError;
205 angle = (angle > m_ptrOffScaleAngle ? m_ptrOffScaleAngle : angle);
206 angle = (angle < -m_ptrOffScaleAngle ? -m_ptrOffScaleAngle : angle);
207 int ptrLength = 0.975*m_dialRadius;
208 int ptrEndX = m_ptrOriginX + (ptrLength*sin(angle*pi/180));
209 int ptrEndY = m_ptrOriginY - (ptrLength*cos(angle*pi/180));
210 pen.SetWidth(3);
211 guiMeterDC.SetPen(pen);
212 guiMeterDC.DrawLine(m_ptrOriginX, m_ptrOriginY, ptrEndX, ptrEndY);
213 }
214
215 // Indicator dots
216 pen.SetWidth(1);
217 guiMeterDC.SetPen(pen);
218
219 if (m_voiced) {
220 int redBelow = -m_ptrMaxError/3;
221 int greenAbove = redBelow;
222 int greenBelow = -redBelow;
223 int redAbove = -redBelow;
224
225 if (m_pitchError<redBelow) {
226 guiMeterDC.SetBrush(*wxRED_BRUSH);
227 guiMeterDC.DrawCircle(m_dotLeftX, m_dotY, m_dotRadius);
228 guiMeterDC.SetBrush(*wxWHITE_BRUSH);
229 guiMeterDC.DrawCircle(m_dotRightX, m_dotY, m_dotRadius);
230 } else if (m_pitchError>greenAbove && m_pitchError<greenBelow) {
231 guiMeterDC.SetBrush(*wxGREEN_BRUSH);
232 guiMeterDC.DrawCircle(m_dotLeftX, m_dotY, m_dotRadius);
233 guiMeterDC.DrawCircle(m_dotRightX, m_dotY, m_dotRadius);
234 } else if (m_pitchError>redAbove) {
235 guiMeterDC.SetBrush(*wxWHITE_BRUSH);
236 guiMeterDC.DrawCircle(m_dotLeftX, m_dotY, m_dotRadius);
237 guiMeterDC.SetBrush(*wxRED_BRUSH);
238 guiMeterDC.DrawCircle(m_dotRightX, m_dotY, m_dotRadius);
239 }
240 } else {
241 guiMeterDC.SetBrush(*wxWHITE_BRUSH);
242 guiMeterDC.DrawCircle(m_dotLeftX, m_dotY, m_dotRadius);
243 guiMeterDC.DrawCircle(m_dotRightX, m_dotY, m_dotRadius);
244 }
