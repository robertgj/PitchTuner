130 std::size_t saOutputStreamImpl::Write( std::vector<saSample>& buffer,
131 const std::size_t framesToWrite )
132 {
133 // Initialise
134 std::size_t samplesWrittenToFragment = 0;
135 std::size_t samplesSpaceRemainingInFragment = 0;
136 if (currentFragment != 0)
137 {
138 samplesSpaceRemainingInFragment =
139 samplesPerFrame*framesSpaceRemainingInFragment;
140 samplesWrittenToFragment = samplesPerFrame*framesWrittenToFragment;
141 }
142
143 // Write. Copy samples from the buffer into a fragment Collect'ed from
144 // the device. As the samples are copied they are are transformed to the
145 // saSinkType. When the fragment is full, or there are no more samples,
146 // Dispatch the fragment to the device.
147 std::size_t samplesWritten = 0;
148 std::size_t samplesLeftToWrite = samplesPerFrame*framesToWrite;
149 while (samplesLeftToWrite > 0)
150 {
151 if (samplesSpaceRemainingInFragment >= samplesLeftToWrite)
152 {
153 std::transform(buffer.begin()+static_cast<long>(samplesWritten),
154 buffer.begin()+
155 static_cast<long>(samplesWritten+samplesLeftToWrite),
156 currentFragment+samplesWrittenToFragment,
157 saSampleToSink<saOutputSink::saSinkType>());
158 samplesWrittenToFragment += samplesLeftToWrite;
159 samplesWritten += samplesLeftToWrite;
160 samplesSpaceRemainingInFragment -= samplesLeftToWrite;
161 samplesLeftToWrite = 0;
162 }
163 else if (samplesSpaceRemainingInFragment > 0)
164 {
165 std::transform(buffer.begin()+static_cast<long>(samplesWritten),
166 buffer.begin()+static_cast<long>(samplesWritten)+
167 static_cast<long>(samplesSpaceRemainingInFragment),
168 currentFragment+samplesWrittenToFragment,
169 saSampleToSink<saOutputSink::saSinkType>());
170 samplesWrittenToFragment += samplesSpaceRemainingInFragment;
171 samplesWritten += samplesSpaceRemainingInFragment;
172 samplesLeftToWrite -= samplesSpaceRemainingInFragment;
173 samplesSpaceRemainingInFragment = 0;
174 }
175 else
176 {
177 if (currentFragment != 0)
178 {
179 outputSink->Dispatch(currentFragment,
180 samplesWrittenToFragment/samplesPerFrame);
181 }
182 // Get a memory fragment from the sink device.
183 // The output samples will be copied into this fragment
184 currentFragment = outputSink->Claim();
185 samplesWrittenToFragment = 0;
186 samplesSpaceRemainingInFragment = samplesPerFrame*framesPerFragment;
187 }
188 }
189
190 // Done
191 framesSpaceRemainingInFragment =
192 samplesSpaceRemainingInFragment/samplesPerFrame;
193 framesWrittenToFragment = samplesWrittenToFragment/samplesPerFrame;
194 std::size_t framesWritten = samplesWritten/samplesPerFrame;
195 framesWrittenToStream += framesWritten;
196 return framesWritten;
197 }
