140 saInputSource::saInputFragment
141 saInputFileWav::Collect(std::size_t& framesRead)
142 {
143 // Read
144 char* buf = reinterpret_cast<char*>(&(sourceBuffer[0]));
145 inputFile.read(buf, static_cast<long>(bytesPerFragment));
146 if ( !inputFile.good() && !inputFile.eof() )
147 {
148 throw std::runtime_error("read from file failed");
149 }
150
151 // Zero out bytes in fragment not filled
152 std::size_t bytesRead = static_cast<std::size_t>(inputFile.gcount());
153 std::size_t samplesRead = bytesRead/static_cast<std::size_t>(bytesPerSample);
154 if ( samplesRead < samplesPerFragment )
155 {
156 memset(&(sourceBuffer[samplesRead]), 0, bytesPerFragment-bytesRead);
157 }
158
159 // Swap bytes for big-endian
160 if ( !saIsLittleEndian() )
161 {
162 std::for_each(sourceBuffer.begin(),
163 sourceBuffer.begin()+static_cast<long>(samplesRead),
164 saSwapSample<saInputSource::saSourceType>());
165 }
166
167 // Set frames read
168 framesRead = samplesRead/samplesPerFrame;
169 framesReadFromSource += framesRead;
170
171 // Done
172 return &(sourceBuffer[0]);
173 }

175 saInputSource* saInputSourceOpenFileWav(const std::string& fileName)
176 {
177 // Open the input file
178 std::string name(fileName.begin(), fileName.end());
179 std::ifstream inputFile(name.c_str(), std::ios::binary);
180 if ( !inputFile.good() )
181 {
182 throw std::runtime_error("can't open source file");
183 }
184
185 // Read the header
186 saWavHeader wavHeader;
187 inputFile.read((char*)&wavHeader, sizeof(saWavHeader));
188 if ( !inputFile.good() )
189 {
190 throw std::runtime_error(" failed read wav header from source file");
191 }
192 if ( !saIsLittleEndian() )
193 {
194 saSwapWavHeader(&wavHeader);
195 }
196
197 // Close the file
198 inputFile.close();
199
200 // Sanity check on IDs
201 if ( ! (saCompareID(wavHeader.RiffChunkID, "RIFF") &&
202 saCompareID(wavHeader.RiffFormat, "WAVE") &&
203 saCompareID(wavHeader.FormatChunkID, "fmt ") &&
204 saCompareID(wavHeader.DataChunkID, "data")))
205 {
206 throw std::runtime_error("id failed in source file");
207 }
208
209 // Sanity check on content
210 if (wavHeader.FormatTag != 1)
211 {
212 throw std::runtime_error("compression not supported");
213 }
214
215 // Sanity check
216 if ((wavHeader.FormatChannels*wavHeader.FormatBitsPerSample/8) >=
217 wavHeader.DataChunkSize/2)
218 {
219 std::cerr << "Sanity check: Did you expect "
220 << wavHeader.FormatChannels
221 << " channels per frame?"
222 << std::endl;
223 }
224
225 // Select source type
226 switch (wavHeader.FormatBitsPerSample)
227 {
228 case 16:
229 return new saInputFileWav(fileName, &wavHeader);
230 break;
231
232 default:
233 throw std::runtime_error("source bits per sample not supported");
234 break;
235 }
236
237 // Done
238 return 0;
239 }
