137static inline void
138HighPassButterworthWcThetaToPqd(const float wc, const float theta, pqd_t &pqd)
139{
140 float lambda = cosf(theta);
141 float k1 = wc*wc;
142 float k2 = 2*wc*lambda;
143 float d = 1/(1+k1-k2);
144 float p1 = 2*(k1-1)*d;
145 float p2 = (1+k1+k2)*d;
146 float q1 = (-2-p1)*d;
147 float q2 = (1-p2)*d;
148
149 pqd = { p1, p2, q1, q2, d };
150}

154static void
155LowPassButterworthWcThetaToPqd(const float wc, const float theta, pqd_t &pqd)
156{
157 float lambda = cosf(theta);
158 float k1 = wc*wc;
159 float k2 = 2*wc*lambda;
160 float d = k1/(1+k1-k2);
161 float p1 = 2*(k1-1)/(1+k1-k2);
162 float p2 = (1+k1+k2)/(1+k1-k2);
163 float q1 = (2-p1)*d;
164 float q2 = (1-p2)*d;
165
166 pqd = { p1, p2, q1, q2, d };
167}

171static void
172PqdToLowNoiseAbcd(const pqd_t pqd, Abcd_t &Abcd)
173{
174 // Design equations for a minimum noise 2nd order state variable section
175 float p1 = pqd.p1;
176 float p2 = pqd.p2;
177 float q1 = pqd.q1;
178 float q2 = pqd.q2;
179 float d = pqd.d;
180
181 float v1 = q2/q1;
182 float v2 = sqrtf((v1*v1)-(p1*v1)+p2);
183 float v3 = v1-v2;
184 float v4 = v1+v2;
185 float v5 = p2-1;
186 float v6 = p2+1;
187 float v7 = v5*((v6*v6)-(p1*p1));
188 float v8 = (p1*p1/4)-p2;
189
190 float b1 = sqrtf(v7/((2*p1*v3) - (v6*(1+(v3*v3)))));
191 float b2 = sqrtf(v7/((2*p1*v4) - (v6*(1+(v4*v4)))));
192 float a21 = sqrtf((((b2*b2)+v5)*v8)/((b1*b1)+v5));
193 float a11 = -p1/2;
194 float a22 = a11;
195 float a12 = v8/a21;
196 float c1 = q1/(2*b1);
197 float c2 = q1/(2*b2);
198
199 Abcd = { a11, a12, a21, a22, b1, b2, c1, c2, d };
200}

205template <typename T>
206class StateVariable2ndOrderSection : public Filter<T>
207{
208public:
209 StateVariable2ndOrderSection () { }
210
211 ~StateVariable2ndOrderSection() throw() {}
212
215 void Init (const Abcd_t Abcd)
216 {
217 a11 = Abcd.a11; a12 = Abcd.a12; a21 = Abcd.a21; a22 = Abcd.a22;
218 b1 = Abcd.b1; b2 = Abcd.b2; c1 = Abcd.c1; c2 = Abcd.c2; d = Abcd.d;
219 x1 = 0; x2 = 0;
220 }
221
225 T operator()(T& u)
226 {
227 // State variable update
228 T x1n = (a11*x1) + (a12*x2) + (b1*u);
229 T x2n = (a21*x1) + (a22*x2) + (b2*u);
230 T y = (c1*x1) + (c2*x2) + (d*u);
231 x1 = x1n;
232 x2 = x2n;
233 return y;
234 }
235
236private:
238 T a11, a12, a21, a22, b1, b2, c1, c2, d;
239
241 T x1, x2;

305 ButterworthHighPass2ndOrderFilter(float cutoffFrequency,

307 {
308 // pi
309 const float pi = std::numbers::pi_v<float>;
310
311 // Warp the frequency scale

402template <typename T>
403class ButterworthHighPass3rdOrderFilter : public Filter<T>
404{
405public:
409 ButterworthHighPass3rdOrderFilter(float cutoffFrequency,
410 std::size_t sampleRate)
411 {
412 // pi
413 const float pi = std::numbers::pi_v<float>;
414
415 // Warp the frequency scale
416 float wc = tanf(static_cast<float>(cutoffFrequency) * pi /
417 static_cast<float>(sampleRate));
418
419 // Get the Butterworth pole positions in the upper left s-plane quadrant
420 // For a 3rd order filter use a cascade of first and second order sections
421 // with poles at 2*pi/3, 4*pi/3 and pi on the s-plane wc circle.
422 float theta = 2*pi/3;
423
424 // Implement the filter as the parallel combination of Gray and Markel
425 // GM2 and GM1 all-pass filters
426
427 // Initialise Gray and Markel GM2 second order section
428
429 // Find the denominator polynomial a0 + a1/z + a2/(z^2)
430 float lambda = cosf(theta);
431 float a0=1-(2*wc*lambda)+(wc*wc);
432 float a1=2*((wc*wc)-1);
433 // float a2=1+(2*wc*lambda)+(wc*wc); // Not used!
434 float re_r2=-a1/(2*a0);
435 float im_r2=(4*wc*sinf(theta))/(2*a0);
436 float abs_r2=sqrtf((re_r2*re_r2)+(im_r2*im_r2));
437 float arg_r2=atan2f(im_r2,re_r2);
438
439 // Find the coefficients of the Gray and Markel GM2 all-pass filter
440 float B1=-2*abs_r2*cosf(arg_r2);
441 float B2=abs_r2*abs_r2;
442 float k2=B2;
443 float e1=-1;
444 float k1=B1/(1+B2);
445 float e2=-1;
446
447 // Convert the Gray and Markel GM2 all-pass filter section to state
448 // variable form
449 float a11=-k1;
450 float a12=(k1*e1)+1;
451 float a21=((k1*e1)-1)*k2;
452 float a22=-k2*k1;
453 float b1=0;
454 float b2=(k2*e2)+1;
455 float c1=((e1*k1-1)*(e2*k2))-(e1*k1)+1;
456 float c2=k1*(1-(e2*k2));
457 float d=k2;
458 Abcd_t Abcd = {a11,a12,a21,a22,b1,b2,c1,c2,d};
459 f2.Init(Abcd);
460
461 // Initialise Gray and Markel GM1 first order section
462
463 // Find the root of the 1st order denominator polynomial 1-r1/z
464 float r1=(1-wc)/(1+wc);
465
466 // Find the coefficient of the Gray and Markel GM1 filter
467 float k=-r1;
468 float e=-1;
469
470 // Convert the Gray and Markel GM1 all-pass filter section to state
471 // variable form
472 float a=-k;
473 float b=1+(k*e);
474 float c=1-(k*e);
475 float dd=k;
476 f1.Init(a,b,c,dd);
477 }
478
479 ~ButterworthHighPass3rdOrderFilter() throw() { }
480
485 T operator()(T& u)
486 {
487 T y1 = f1(u);
488 T y2 = f2(u);
489 T y = (y2-y1)/2;
490
491 return y;
492 }
493
494private:
496 StateVariable1stOrderSection<T> f1;
497 StateVariable2ndOrderSection<T> f2;
498};

505template <typename T>
506class ButterworthHighPassStoyanov3rdOrderFilter : public Filter<T>
507{
508public:
512 ButterworthHighPassStoyanov3rdOrderFilter(float cutoffFrequency,
513 std::size_t sampleRate)
514 {
515 // pi
516 const float pi = std::numbers::pi_v<float>;
517
518 // Warp the frequency scale
519 float wc = tanf(static_cast<float>(cutoffFrequency) * pi /
520 static_cast<float>(sampleRate));
521
522 // Get the Butterworth pole positions in the upper left s-plane quadrant
523 // For a 3rd order filter use a cascade of first and second order sections
524 // with poles at 2*pi/3, 4*pi/3 and pi on the s-plane wc circle.
525 float theta = 2*pi/3;
526
527 // Implement the filter as the parallel combination of
528 // Stoyanov LS2a and LS1 all-pass filters
529
530 // Initialise Stoyanov LS2a second order section
531
532 // Find the denominator polynomial a0 + a1/z + a2/(z^2)
533 float lambda = cosf(theta);
534 float a0=1-(2*wc*lambda)+(wc*wc);
535 float a1=2*((wc*wc)-1);
536 // float a2=1+(2*wc*lambda)+(wc*wc); // Not used!
537 float re_r2=-a1/(2*a0);
538 float im_r2=(4*wc*sinf(theta))/(2*a0);
539 float abs_r2=sqrtf((re_r2*re_r2)+(im_r2*im_r2));
540 float arg_r2=atan2f(im_r2,re_r2);
541
542 // Find the coefficients of the Stoyanov LS2a all-pass filter
543 float B1=-2*abs_r2*cosf(arg_r2);
544 float B2=abs_r2*abs_r2;
545 float C2=1-B2;
546 float C1=(B1+2-C2)/2;
547
548 // Convert the Stoyanov LS2a all-pass filter section to state variable form
549 float a11 = 1-C1;
550 float a12 = -C1;
551 float a21 = -C2-C1+2;
552 float a22 = -C2-C1+1;
553 float b1 = C1;
554 float b2 = C2+C1-2;
555 float c1 = C2;
556 float c2 = C2;
557 float d = 1-C2;
558 Abcd_t Abcd = {a11,a12,a21,a22,b1,b2,c1,c2,d};
559 f2.Init(Abcd);
560
561 // Initialise Stoyanov LS1 first order section
562
563 // Find the root of the 1st order denominator polynomial 1-r1/z
564 float r1 = (1-wc)/(1+wc);
565
566 // Find the coefficient of the Stoyanov LS1 filter
567 float C = 1-r1;
568
569 // Convert the Stoyanov LS1 all-pass filter section to state variable form
570 // (Note the missing - at the left of the output in Stoyanov's Figure 3(b).)
571 float a = 1-C;
572 float b = C;
573 float c = 2-C;
574 float dd = C-1;
575 f1.Init(a,b,c,dd);
576 }
577
578 ~ButterworthHighPassStoyanov3rdOrderFilter() throw() { }
579
584 T operator()(T& u)
585 {
586 T y1 = f1(u);
587 T y2 = f2(u);
588 T y = (y2-y1)/2;
589
590 return y;
591 }
592
593private:
595 StateVariable1stOrderSection<T> f1;
596 StateVariable2ndOrderSection<T> f2;
597};
