165 void saOutputFileWav::Dispatch(saOutputFragment fragment,
166 const std::size_t validFrames)
167 {
168 // Swap bytes for big-endian
169 if ( !saIsLittleEndian() )
170 {
171 std::for_each(&(fragment[0]),
172 &(fragment[samplesPerFrame*validFrames]),
173 saSwapSample<saSinkType>());
174 }
175
176 // Write
177 char* buf = reinterpret_cast<char*>(&(fragment)[0]);
178 outputFile.write(buf,
179 static_cast<std::streamsize>(validFrames*bytesPerFrame));
180 if (!outputFile.good())
181 {
182 throw std::runtime_error(" failed writing fragment");
183 }
184 framesWrittenToSink += validFrames;
185
186 return;
187 }

189 void saOutputFileWav::WriteWavFileHeader()
190 {
191 // Fill in sink parameters
192
193 // Format
194 memcpy(&(wavHeader.FormatChunkID), "fmt ", 4);
195 wavHeader.FormatChunkSize = 16;
196 wavHeader.FormatTag = 1;
197 wavHeader.FormatChannels = (short unsigned int)samplesPerFrame;
198 wavHeader.FormatBlockAlign = (short unsigned int)bytesPerFrame;
199 wavHeader.FormatAvgBytesPerSec =
200 wavHeader.FormatBlockAlign*((unsigned int)framesPerSecond);
201 wavHeader.FormatSamplesPerSec = (short unsigned int)framesPerSecond;
202 wavHeader.FormatBitsPerSample = (short unsigned int)(bytesPerSample*8);
203
204 // Data
205 memcpy(&(wavHeader.DataChunkID), "data", 4);
206 wavHeader.DataChunkSize =
207 wavHeader.FormatBlockAlign*(int)framesWrittenToSink;
208
209 // RIFF
210 memcpy(&(wavHeader.RiffChunkID), "RIFF", 4);
211 wavHeader.RiffChunkSize = 4+(8+wavHeader.FormatChunkSize)+
212 (8+wavHeader.DataChunkSize);
213 memcpy(&(wavHeader.RiffFormat), "WAVE", 4);
214
215 // Big-endian?
216 if ( !saIsLittleEndian() )
217 {
218 saSwapWavHeader(&wavHeader);
219 }
220
221 // Rewind output file
222 outputFile.seekp(std::ios_base::beg);
223 if ( !outputFile.good() )
224 {
225 throw std::runtime_error(" failed seeking to start of sink file");
226 }
227
228 // Write output file header
229 char *buf = reinterpret_cast<char*>(&wavHeader);
230 outputFile.write(buf, sizeof(saWavHeader));
231 if ( !outputFile.good() )
232 {
233 throw std::runtime_error(" failed writing wav header to sink file");
234 }
235 }
