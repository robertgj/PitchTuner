246{

249 // Register file handlers

251 saSingletonInputFileSelector::Instance().

259 // Default arguments
260 bool helpReq = false;
261 bool debugReq = false;
262 std::size_t maxPitchSamples = 0;
263 std::size_t channel = 0;
264 std::size_t sampleRate = 10000;
265 std::size_t msDeviceLatency = 100;
266 std::size_t msWindow = 25;
267 std::size_t msTmax = 20;
268 std::size_t msTsample = 10;
269 std::size_t subSample = 1;
270 float inputLpFilterCutoff = 1000;
271 float baseLineHpFilterCutoff = 200;
272 bool disableHpFilter = false;
273 bool disableAgc = false;
274 bool removeDC = false;
275 float threshold = 0.1f;
276 std::string sourceName("default");
277 bool fileReq = false;
278 bool maxPitchSamplesSet = false;
279
280 // Parse arguments
281 ParseOptions(argc, argv, helpReq, debugReq,
282 maxPitchSamples, channel, sampleRate,
283 msDeviceLatency, msWindow, msTmax, msTsample, subSample,
284 inputLpFilterCutoff, baseLineHpFilterCutoff,
285 disableHpFilter, disableAgc,
286 threshold, removeDC, sourceName, fileReq, maxPitchSamplesSet);
287
288 // Echo PitchTracker arguments
289 if (debugReq)
290 {
291 std::cerr << "debug= " << debugReq << std::endl;
292 std::cerr << "maxPitchSamples= " << maxPitchSamples << std::endl;
293 std::cerr << "channel= " << channel << std::endl;
294 std::cerr << "sampleRate= " << sampleRate << std::endl;
295 std::cerr << "msDeviceLatency= " << msDeviceLatency << std::endl;
296 std::cerr << "msWindow= " << msWindow << std::endl;
297 std::cerr << "msTmax= " << msTmax << std::endl;
298 std::cerr << "msTsample= " << msTsample << std::endl;
299 std::cerr << "subSample= " << subSample << std::endl;
300 std::cerr << "inputLpFilterCutoff= "
301 << inputLpFilterCutoff << std::endl;
302 std::cerr << "baseLineHpFilterCutoff= "
303 << baseLineHpFilterCutoff << std::endl;
304 std::cerr << "disableHpFilter= " << disableHpFilter << std::endl;
305 std::cerr << "disableAgcFilter= " << disableAgc << std::endl;
306 std::cerr << "removeDC= " << removeDC << std::endl;
307 std::cerr << "threshold= " << threshold << std::endl;
308 std::cerr << "sourceName= " << sourceName << std::endl;
309 }
310
311 // Check for help request
312 if (helpReq)
313 {
314 return 0;
315 }
316
317 // Open the input stream
318 std::string name(sourceName.begin(), sourceName.end());
319 std::unique_ptr<saInputStream>
320 is( fileReq == true ? saInputStreamOpenFile(name) :
321 saInputStreamOpenDevice
322 (name, sampleRate, msDeviceLatency, 1, debugReq) );
323 if ( fileReq )
324 {
325 sampleRate = is->GetFramesPerSecond();
326
327 if ( !maxPitchSamplesSet )
328 {
329 maxPitchSamples = is->GetFramesPerStream();
330 maxPitchSamples /= (msTsample*sampleRate/1000);
331 }
332 }
333
334 // Create PreProcessor
335 std::unique_ptr<PreProcessor>
336 pre(PreProcessorCreate(is.get(), channel,
337 msWindow, msTmax, msTsample, subSample,
338 inputLpFilterCutoff, baseLineHpFilterCutoff,
339 disableHpFilter, disableAgc, debugReq));
340
341 // Create PitchTracker
342 std::unique_ptr<PitchTracker>
343 pt( PitchTrackerCreate
344 ( pre->SubSampleRate(), pre->WindowLength(), pre->MaxLags(),
345 threshold, removeDC, debugReq ) );
346
347 std::vector<float> pitchList;
348 std::size_t samplesSoFar = 0;
349 while( samplesSoFar++ < maxPitchSamples && !is->EndOfSource() )
350 {
351 float pitch = pt->EstimatePitch( pre->Read() );
352 pitchList.push_back(pitch);
353 }
354
355 // Echo pitch results
356 std::cout.precision(4);
357 std::copy(pitchList.begin(), pitchList.end(),
358 std::ostream_iterator<float>(std::cout, "\n"));
359
360 // Done. Assume destructor closes stream
361 }
362 catch(std::exception& excpt)
363 {
364 std::cerr << excpt.what() << std::endl ;
365 return -1;
366 }
367
368 return 0;
369}
