229 saOutputDevice::~saOutputDevice( ) throw()
230 {
231 try
232 {
233 // Close audio device
234
235 int err = snd_pcm_drain(handle);
236 if (err == -ESTRPIPE)
237 {
238 throw std::runtime_error
239 (std::string("audio output device write failed: ") +
240 std::string("a suspend event occurred") );
241 }
242 else if (err < 0)
243 {
244 throw std::runtime_error
245 (std::string("audio output device write failed"));
246 }
247
248 snd_pcm_close(handle);
249 snd_config_update_free_global();
250 }
251 catch(std::exception& excpt)
252 {
253 std::cerr << excpt.what() << std::endl ;
254 }

262 static void saOutputDeviceWrite(snd_pcm_t *handle,
263 saOutputSink::saSinkType* buffer,
264 const std::size_t framesToWrite,
265 bool &isUnderRun)
266 {
267 long numWritten = snd_pcm_writei(handle, buffer, framesToWrite);
268
269 int err = (int)numWritten;
270 if (err == -EPIPE)
271 {
272 numWritten = 0;
273 isUnderRun = true;
274 }
275 else if (err == -EBADFD)
276 {
277 numWritten = 0;
278 throw std::runtime_error
279 ( std::string("audio output device write failed: ") +
280 std::string("PCM is not in the right state") );
281 }
282 else if (err == -ESTRPIPE)
283 {
284 numWritten = 0;
285 throw std::runtime_error
286 ( std::string("audio output device write failed: ") +
287 std::string("a suspend event occurred") );
288 }
289 else if (err < 0)
290 {
291 numWritten = 0;
292 throw std::runtime_error
293 ( std::string("audio output device write failed: ") +
294 std::string(snd_strerror(err)));
295 }
296 else if (static_cast<std::size_t>(numWritten) != framesToWrite)
297 {
298 throw std::runtime_error
299 ( "audio output device write incomplete!" );
300 }
301 }
