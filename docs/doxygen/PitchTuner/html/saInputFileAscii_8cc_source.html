160 saInputFileAscii::saInputFileAscii(const std::string fileName,
161 const std::size_t _framesPerSecond,
162 const std::size_t _samplesPerFrame,
163 const std::size_t _framesPerSource)
164 : sourceName( fileName.begin(), fileName.end() ),
165 framesPerSecond( _framesPerSecond ),
166 samplesPerFrame( _samplesPerFrame ),
167 framesPerFragment( 16384 ), // Arbitrary size!
168 samplesPerFragment( samplesPerFrame*framesPerFragment ),
169 framesPerSource( _framesPerSource ),
170 framesReadFromSource( 0 ),
171 sourceBuffer( samplesPerFragment ),
172 rff( sourceName.c_str() )
173 {
174 if ( !rff.is_ok() )
175 {
176 throw std::runtime_error("can't open source file");
177 }
178
179 // Read past the header
180 std::size_t header[3];
181 if(!rff.read_header(header))
182 {
183 throw std::runtime_error("read of header failed");
184 }
185
186 // Sanity check
187 if((header[0] != framesPerSecond) ||
188 (header[1] != samplesPerFrame) ||
189 (header[2] != framesPerSource))
190 {
191 throw std::runtime_error("inconsistent header");
192 }
193 }

200 saInputFileAscii::Collect(std::size_t& framesRead)
201 {
202 // Initialise
203 std::size_t framesRemainingInSource = framesPerSource-framesReadFromSource;
204 std::size_t framesToRead = framesRemainingInSource < framesPerFragment ?
205 framesRemainingInSource : framesPerFragment;
206 std::size_t samplesToRead = samplesPerFrame*framesToRead;
207
208
209 // Read
210 sourceBuffer.resize(0);
211 for (std::size_t i = 0; i < samplesToRead; i++)
212 {
213 sourceBuffer[i] = rff.read();
214 }
215
216 // Check
217 if (!rff.is_ok())
218 {
219 framesRemainingInSource = 0;
220 throw std::runtime_error("read from file failed");
221 }
222
223 // Set frames read
224 framesReadFromSource += framesToRead;
225 framesRead = framesToRead;
226
227 // Done
228 return (framesRead > 0 ? &(sourceBuffer[0]) : 0);
229 }

231 saInputSource* saInputSourceOpenFileAscii(const std::string& fileName)
232 {
233 // Open the input file
234 std::string name( fileName.begin(), fileName.end() );
235 std::ifstream inputFile( name.c_str() );
236 if ( !inputFile.good() )
237 {
238 throw std::runtime_error("can't open source file");
239 }
240
241 // Read the header
242 std::size_t header[3];
243 for (std::size_t i = 0; i < 3; i++)
244 {
245 header[i] = *std::istream_iterator<std::size_t>(inputFile);
246 if ( inputFile.fail() )
247 {
248 throw std::runtime_error("header read failed");
249 }
250 }
251 inputFile.close();
252
253 return new saInputFileAscii(fileName, header[0], header[1], header[2]);
254 }
