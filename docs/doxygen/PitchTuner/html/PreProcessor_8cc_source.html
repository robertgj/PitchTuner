33template <typename T>
34struct DecimationFilter
35{
36 DecimationFilter(std::vector<T>& _u, Filter<T>& _f, std::size_t _dec)
37 : u(_u), f(_f), dec(_dec), base(0) { }
38
41 T operator() ()
42 {
43 T val = 0;
44 for(std::size_t i=0; i<dec; i++)
45 {
46 val = f(u[base++]);
47 }
48 return val;
49 }
50
51private:
52 std::vector<T>& u;
53 Filter<T>& f;
54 std::size_t dec;
55 std::size_t base;
56};

258const std::deque<saSample>& PreProcessorImpl::Read( )
259{
260 // Check for over-run
261 if ( inputStream->OverRun() )
262 {
263 // Clear
264 inputStream->Clear();
265 outputDeque.assign(subSampledOutputSize, 0);
266 if (debug)
267 {
268 std::cerr << "Stream reset after overrun\n";
269 }
270 }
271
272 // Discard oldest elements
273 if (outputDeque.size() >= subSampledInterval)
274 {
275 outputDeque.erase(outputDeque.begin(),
276 outputDeque.begin()+
277 static_cast<long>(subSampledInterval));
278 }
279 else if ( outputDeque.size() > 0)
280 {
281 // Should not get here!
282 outputDeque.resize(0);
283 }
284
285 // Loop filling the deque
286 while (outputDeque.size() < subSampledOutputSize)
287 {
288 // Read from stream
289 y.resize(0);
290 std::size_t numRead = inputStream->Read( y, sampleInterval, channel );
291 if (numRead != sampleInterval)
292 {
293 // Sanity check
294 if ( !inputStream->EndOfSource() )
295 {
296 inputStream->Clear();
297 outputDeque.assign(subSampledOutputSize, 0);
298 if (debug)
299 {
300 std::cerr << "Stream reset after under-run\n";
301 }
302 return outputDeque;
303 }
304 // Pad outputDeque to required size
305 y.insert(y.end(), sampleInterval-y.size(), 0);
306 }
307
308 // Do low-pass filter
309 ylp.resize(0);
310 if (subSample > 1)
311 {
312 std::generate_n(std::back_inserter(ylp), subSampledInterval,
313 DecimationFilter<saSample>(y, lpFilter, subSample));
314 }
315 else
316 {
317 std::transform(y.begin(), y.end(), std::back_inserter(ylp),
318 updateFilter<saSample>(&lpFilter));
319 }
320
321 if ( disableHpFilter && disableAgc )
322 {
323 // Copy to deque
324 outputDeque.insert(outputDeque.end(), ylp.begin(), ylp.end());
325 }
326 else if ( (!disableHpFilter) && disableAgc )
327 {
328 // Do high-pass filter
329 yhp.resize(0);
330 std::transform(ylp.begin(), ylp.end(), std::back_inserter(yhp),
331 updateFilter<saSample>(&hpFilter));
332
333 // Copy to deque
334 outputDeque.insert(outputDeque.end(), yhp.begin(), yhp.end());
335 }
336 else if ( disableHpFilter && (!disableAgc) )
337 {
338 // Do AGC
339 yagc.resize(0);
340 std::transform(ylp.begin(), ylp.end(), std::back_inserter(yagc),
341 updateFilter<saSample>(&agc));
342
343 // Copy to deque
344 outputDeque.insert(outputDeque.end(), yagc.begin(), yagc.end());
345 }
346 else
347 {
348 // Do high-pass filter
349 yhp.resize(0);
350 std::transform(ylp.begin(), ylp.end(), std::back_inserter(yhp),
351 updateFilter<saSample>(&hpFilter));
352
353 // Do AGC
354 yagc.resize(0);
355 std::transform(yhp.begin(), yhp.end(), std::back_inserter(yagc),
356 updateFilter<saSample>(&agc));
357
358 // Copy to deque
359 outputDeque.insert(outputDeque.end(), yagc.begin(), yagc.end());
360 }
361
362 // Save
363 if ( debug )
364 {
365 yList.insert(yList.end(), y.begin(), y.end());
366 ylpList.insert( ylpList.end(), ylp.begin(), ylp.end() );
367 yhpList.insert( yhpList.end(), yhp.begin(), yhp.end() );
368 yagcList.insert( yagcList.end(), yagc.begin(), yagc.end() );
369 }
370 }
371
372 return outputDeque;
373}
